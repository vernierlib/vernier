47  double FingerprintPatternLayout::getIntensity(double x, double y) {
48  int col = std::round((x + originX) / dotSize - 0.5);
49  int row = std::round((y + originY) / dotSize - 0.5);
50 
51  if (col < 0 || row < 0 || col >= 2 * bitmap.cols() || row >= 2 * bitmap.rows()) {
52  return 0;
53  } else if (row % 2 == 1 || col % 2 == 1) {
54  return 0;
55  } else {
56  double periodicIntensity = (1 + cos(2 * PI * x / dotSize)) * (1 + cos(2 * PI * y / dotSize)) / 4;
57  return bitmap(row / 2, col / 2) * periodicIntensity;
58  }
59  }
