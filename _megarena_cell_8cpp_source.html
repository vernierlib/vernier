19  Eigen::Array33d countOnlyDots = Eigen::Array33d::Zero(3, 3);
20 
21  for (int col = 0; col < thumbnailNumberWhiteDots.cols(); col++) {

23  int index1 = row % 3;
24  int index2 = col % 3;
