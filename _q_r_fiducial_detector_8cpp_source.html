137  void QRFiducialDetector::findRowPatterns() {
138  rowPatterns.clear();
139  int pixelIndex = 0;
140  for (int y = 0; y < cannyImage.rows; y++) {
141  int egdeCount = 0;
142  int insideEdge = false;
143  int edgeStart;
144  QRRowPattern rowPattern(y);
145  for (int x = 0; x < cannyImage.cols - 1; x++) {
146  if (cannyImage.data[pixelIndex] > 0 && !insideEdge) { // entering an edge
147  insideEdge = true;
148  edgeStart = x;
149  } else if (cannyImage.data[pixelIndex] == 0 && cannyImage.data[pixelIndex + 1] == 0 && insideEdge) { // exiting an edge
150  insideEdge = false;
151  rowPattern.pushCol(((x - 1) + edgeStart) / 2);
152  egdeCount++;
153  if (egdeCount >= 6 && rowPattern.isWellProportioned(proportionToleranceInPixels)) {
154  rowPatterns.push_back(rowPattern);
155  }
156  }
157  pixelIndex++;
158  }
159  pixelIndex++;
160  }
161  }

163  void QRFiducialDetector::findColPatterns() {
164  colPatterns.clear();
165  for (int x = 0; x < cannyImage.cols; x++) {
166  int egdeCount = 0;
167  int pixelIndex = x;
168  bool insideEdge = false;
169  int edgeStart;
170  QRColumnPattern colPattern(x);
171  for (int y = 0; y < cannyImage.rows - 1; y++) {
172  if (cannyImage.data[pixelIndex] > 0 && !insideEdge) { // entering an edge
173  insideEdge = true;
174  edgeStart = y;
175  } else if (cannyImage.data[pixelIndex] == 0 && cannyImage.data[pixelIndex + cannyImage.cols] == 0 && insideEdge) { // exiting an edge
176  insideEdge = false;
177  colPattern.pushRow(((y - 1) + edgeStart) / 2);
178  egdeCount++;
179  if (egdeCount >= 6 && colPattern.isWellProportioned(proportionToleranceInPixels)) {
180  colPatterns.push_back(colPattern);
181  }
182  }
183  pixelIndex += cannyImage.cols;
184  }

201  void QRFiducialDetector::findGroupsOfPatterns() {
202  clearGroups();
203  for (int row = 0; row < rowPatterns.size(); row++) {
204  if (rowPatternGroup[row] < 0) { // row pattern not in a group
205  for (int col = 0; col < colPatterns.size(); col++) {
206  if (colPatternGroup[col] < 0) { // col pattern not in a group
207  if (rowPatterns[row].isCrossing(colPatterns[col])) { // found a new group
208  int groupNumber = groupsOfColPatterns.size();
209  rowPatternGroup[row] = groupNumber;
210  colPatternGroup[col] = groupNumber;
211  groupsOfColPatterns.push_back(std::vector<int>());
212  groupsOfRowPatterns.push_back(std::vector<int>());
213  groupsOfColPatterns[groupNumber].push_back(col);
214  groupsOfRowPatterns[groupNumber].push_back(row);
215 
216  // look for other crossing patterns
217  for (int c = 0; c < colPatterns.size(); c++) {
218  if (colPatternGroup[c] < 0 && rowPatterns[row].isCrossing(colPatterns[c])) { // extending an existing group from row
219  colPatternGroup[c] = groupNumber;
220  groupsOfColPatterns[groupNumber].push_back(c);
221  }
222  }
223 
224  for (int r = 0; r < rowPatterns.size(); r++) {
225  if (rowPatternGroup[r] < 0 && rowPatterns[r].isCrossing(colPatterns[col])) { // extending an existing group from col
226  rowPatternGroup[r] = groupNumber;
227  groupsOfRowPatterns[groupNumber].push_back(r);
228  }
229  }
230  }
231  }
232  }
233  }

237  void QRFiducialDetector::sortFiducials() {
238  fiducials.clear();
239  for (int groupNumber = 0; groupNumber < groupsOfRowPatterns.size(); groupNumber++) {
240  double dotSize = 0.0;
241 
242  // subpixel column calculation
243  double x = 0.0;
244  for (int i = 0; i < groupsOfRowPatterns[groupNumber].size(); i++) {
245  int r = groupsOfRowPatterns[groupNumber][i];
246  x += rowPatterns[r].getCol();
247  dotSize += rowPatterns[r].getDotSize();
248  }
249  x /= groupsOfRowPatterns[groupNumber].size();
250 
251  // subpixel row calculation
252  double y = 0.0;
253  for (int i = 0; i < groupsOfColPatterns[groupNumber].size(); i++) {
254  int c = groupsOfColPatterns[groupNumber][i];
255  y += colPatterns[c].getRow();
256  dotSize += colPatterns[c].getDotSize();
257  }
258  y /= groupsOfColPatterns[groupNumber].size();
259 
260  int patternCount = (groupsOfColPatterns[groupNumber].size() + groupsOfRowPatterns[groupNumber].size());
261  dotSize /= patternCount;
262  if (patternCount >= 2 * dotSize) {
263  fiducials.push_back(QRFiducialPattern(cv::Point2d(x, y), patternCount));
264  int i = fiducials.size() - 1;
265  while (i > 0 && fiducials[i - 1].patternCount < fiducials[i].patternCount) {
266  std::swap(fiducials[i], fiducials[i - 1]);
267  i--;
268  }
269  }
270  }
271  }

280  void QRFiducialDetector::compute(const cv::Mat& image) {
281  if (image.channels() > 1) {
282  cv::cvtColor(image, grayImage, cv::COLOR_BGR2GRAY);
283  } else {
284  grayImage = image;
285  }
286 
287  cv::normalize(image, image, 255, 0, cv::NORM_MINMAX);
288  image.convertTo(grayImage, CV_8UC1);
289 
290  //cv::cvtColor(image, grayImage, cv::COLOR_BGR2GRAY);
291  cv::Canny(grayImage, cannyImage, lowCannyThreshold, highCannyThreshold, 3, true);
292  findRowPatterns();
293  findColPatterns();
294  findGroupsOfPatterns();
295  sortFiducials();
296  }
