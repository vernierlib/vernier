32  void HPCodePatternDetector::computeImage() {
33 
34  detector.compute(image8U);
35 
36  markers.clear();
37  for (int i = 0; i < detector.codes.size(); i++) {
38 
39  QRCode code = detector.codes[i];
40 
41  int diameter = (int) (2 * code.getRadius());
42  if (snapshot.cols() < diameter) {
43  std::cout << "The HPCode is too large for pose estimation: increase the snapshot size over " << diameter << " pixels." << std::endl;
44  }
45  if (2 * diameter < numberHalfPeriods) {
46  std::cout << "The HPCode is too tiny for pose estimation: increase the picture quality size." << std::endl;
47  }
48 
49  int centerX = (int) code.center.x;
50  int centerY = (int) code.center.y;
51 
52  takeSnapshot(centerX, centerY, snapshotSize, array, snapshot);
53  patternPhase.compute(snapshot * window);
54  plane1 = patternPhase.getPlane1();
55  plane2 = patternPhase.getPlane2();
56 
57  if (patternPhase.peaksFound()) {
58 
59  double alpha;
60  double dx, dy;
61  double diffAngle = angleInPiPi(plane1.getAngle() - code.getAngle());
62  if (diffAngle >= -PI / 4 && diffAngle <= PI / 4) {
63  alpha = plane1.getAngle();
64  dx = -plane1.getPosition(physicalPeriod, 0.0, 0.0);
65  dy = -plane2.getPosition(physicalPeriod, 0.0, 0.0);
66  } else if (diffAngle >= 3 * PI / 4 || diffAngle <= -3 * PI / 4) {
67  alpha = plane1.getAngle() + PI;
68  dx = +plane1.getPosition(physicalPeriod, 0.0, 0.0);
69  dy = +plane2.getPosition(physicalPeriod, 0.0, 0.0);
70  } else if (diffAngle >= PI / 4 && diffAngle <= 3 * PI / 4) {
71  alpha = plane1.getAngle() - PI / 2;
72  dx = plane2.getPosition(physicalPeriod, 0.0, 0.0);
73  dy = -plane1.getPosition(physicalPeriod, 0.0, 0.0);
74  } else {
75  alpha = plane1.getAngle() + PI / 2;
76  dx = -plane2.getPosition(physicalPeriod, 0.0, 0.0);
77  dy = plane1.getPosition(physicalPeriod, 0.0, 0.0);
78  }
79 
80  double pixelSize = physicalPeriod / patternPhase.getPixelPeriod();
81  double xImg = (centerX - image8U.cols / 2);
82  double yImg = (centerY - image8U.rows / 2);
83  double x = pixelSize * (xImg * cos(alpha) - yImg * sin(-alpha)) + dx;
84  double y = pixelSize * (xImg * sin(-alpha) + yImg * cos(alpha)) + dy;
85 
86  Pose pose = Pose(x, y, alpha, pixelSize);
87 
88  if (numberHalfPeriods % 4 == 1) {
89  unsigned long id = readNumber(code, image8U, patternPhase.getPixelPeriod() / 2.0);
90  markers.insert(std::make_pair(id, pose));
91  } else {
92  markers.insert(std::make_pair(i, pose));
93  }
94  }
95  }

98  unsigned long HPCodePatternDetector::readNumber(QRCode& code, const cv::Mat& image, double dotSize) {
99  cv::Point2d rightDirection = (code.right - code.top);
100  rightDirection *= dotSize / cv::norm(rightDirection);
101  cv::Point2d upDirection = (code.top - code.bottom);
102  upDirection *= dotSize / cv::norm(upDirection);
103  cv::Point2d end = code.top + 3 * upDirection + 3.5 * rightDirection;
104  cv::Point2d start = code.right + 3 * upDirection - 3.5 * rightDirection;
105 
106  // Getting the line
107  cv::LineIterator it(image, cv::Point2i(start), cv::Point2i(end), 8);
108  cv::Mat rasterLine(1, it.count, CV_8U);
109  for (int i = 0; i < it.count; i++, ++it) {
110  rasterLine.at<unsigned char>(0, i) = **it;
111  }
112 
113  // Analysing the line
114  int dotCount = 7 + (int) std::round((cv::norm(code.top - code.right) / dotSize));
115  //std::cout << "dotCount : " << 7 + (double)(cv::norm(code.top - code.right) / dotSize) << std::endl;
116  cv::Mat dataLine(1, dotCount - 14, CV_8U);
117  cv::resize(rasterLine, dataLine, dataLine.size(), 0, 0, cv::INTER_AREA);
118  cv::threshold(dataLine, dataLine, 0, 255, cv::THRESH_BINARY | cv::THRESH_OTSU);
119  //std::cout << "data line : " << dataLine << std::endl;
120  int zeroCount = 0;
121  for (int col = 0; col < dataLine.cols; col += 2) {
122  if (dataLine.at<unsigned char>(0, col) == 0) {
123  zeroCount++;
124  }
125  }
126  int lowLevel = 0;
127  if (zeroCount < dataLine.cols / 4) {
128  lowLevel = 255;
129  }
130 
131  // Reading the data line
132  unsigned long number = 0;
133  int n = 0;
134  for (int col = 1; col < dataLine.cols - 1; col += 2) {
135  if (dataLine.at<unsigned char>(0, col) == lowLevel) {
136  number |= 1 << n;
137  }
138  n++;
139  }
140 
141  return number;
142  }
