20  template<typename _Scalar, int _Rows, int _Cols>
21  void applyBandPassCut(Eigen::Array<_Scalar, _Rows, _Cols>& array, double lowFrequency, double highFrequency) {
22  ASSERT(array.rows() > 0 && array.cols() > 0);
23  int rows = array.rows();
24  int cols = array.cols();
25  int centerX = rows / 2;
26  int centerY = cols / 2;
27 
28  for (int col = 0; col < cols; ++col) {
29  for (int row = 0; row < rows; ++row) {
30  double distance = std::hypot(row - centerX, col - centerY);
31  if (distance <= lowFrequency || distance > highFrequency) {
32  array(row, col) = _Scalar();
33  }
34  }
35  }
36  }

45  void applyAngularCut(Eigen::Array<_Scalar, _Rows, _Cols>& array, double centerAngle, double widthAngle) {
46  ASSERT(array.rows() > 0 && array.cols() > 0);
47  int rows = array.rows();
48  int cols = array.cols();
49  int centerX = rows / 2;
50  int centerY = cols / 2;
51  double halfWidthAngle = widthAngle / 2.0;
52 
53  for (int col = 0; col < cols; ++col) {
54  for (int row = 0; row < rows; ++row) {
55  double currentAngle = std::atan2(row - centerX, col - centerY);
56  double diff = angleInPiPi(currentAngle - centerAngle);
57  if (std::abs(diff) <= halfWidthAngle || std::abs(diff) >= (PI - halfWidthAngle)) {
58  array(row, col) = _Scalar();

71  template<typename _Scalar, int _Rows, int _Cols>
72  void applyGaussianFilter(Eigen::Array<_Scalar, _Rows, _Cols>& array, int centerRow, int centerCol, double sigma) {
73  ASSERT(sigma > 0.0);
74  double invTwoSigmaSq = 1.0 / (2.0 * sigma * sigma);
75 
76  Eigen::ArrayXd rowWeights(array.rows());
77  for (int row = 0; row < array.rows(); row++) {
78  rowWeights(row) = std::exp(-(row - centerRow) * (row - centerRow) * invTwoSigmaSq);
79  }
80 
81  Eigen::ArrayXd colWeights(array.cols());
82  for (int col = 0; col < array.cols(); col++) {
83  colWeights(col) = std::exp(-(col - centerCol) * (col - centerCol) * invTwoSigmaSq);
84  }
85 
86  for (int j = 0; j < array.cols(); j++) {
87  array.col(j) *= rowWeights * colWeights(j);
88  }
89  }

107  for (int row = array.rows() - 1; row >= 0; --row) {
