11  QRCode::QRCode(QRFiducialPattern& marker0, QRFiducialPattern& marker1, QRFiducialPattern& marker2) {
12  // Determining the top marker
13  double d01 = cv::norm(marker0.position - marker1.position);
14  double d12 = cv::norm(marker1.position - marker2.position);
15  double d02 = cv::norm(marker0.position - marker2.position);
16 
17  if (d01 > d02 && d01 > d12) {
18  top = marker2.position;
19  right = marker0.position;
20  bottom = marker1.position;
21  } else if (d12 > d01 && d12 > d02) {
22  top = marker0.position;
23  right = marker1.position;
24  bottom = marker2.position;
25  } else {
26  top = marker1.position;
27  right = marker0.position;
28  bottom = marker2.position;
29  }
30 
31  // Determining the positive sense of rotation
32  cv::Point2d rightDirection = right - top;
33  if (rightDirection.cross(bottom - top) < 0) {
34  std::swap(right, bottom);
35  rightDirection = right - top;
36  }
37 
38  this->center = cv::Point2d((right.x + bottom.x) / 2.0, (right.y + bottom.y) / 2.0);
39  }

63  void QRCodeDetector::computeDistances() {
64  int size = fiducialDetector.fiducials.size();
65  distance.resize(size);
66  for (int i = 0; i < size; i++) {
67  distance[i].resize(size);

70  distance[i][i] = 0.0;

77  void QRCodeDetector::makeFirstClustering() {
78  int clusterCount = fiducialDetector.fiducials.size() / 3;
79  clusters.resize(clusterCount);
80 
81  int markerCount = clusterCount * 3;
82  std::vector<bool> assigned(markerCount);
83  for (int i = 0; i < assigned.size(); i++) {
84  assigned[i] = false;
85  }
86 
87  int currentCluster = 0;
88  while (currentCluster < clusterCount) {
89  clusters[currentCluster].clear();
90  // looking for a not assigned marker
91  int a = 0;
92  while (a < markerCount && assigned[a]) a++;
93 
94  clusters[currentCluster].push_back(a);
95  assigned[a] = true;
96 
97  // looking for a first closest marker
98  int b;
99  double min = 1e150;
100  for (int i = 0; i < markerCount; i++) {
101  if (!assigned[i] && distance[a][i] < min) {
102  b = i;
103  min = distance[a][i];
104  }
105  }
106  clusters[currentCluster].push_back(b);
107  assigned[b] = true;
108 
109  // looking for a second closest marker
110  int c;
111  min = 1e150;
112  for (int i = 0; i < markerCount; i++) {
113  if (!assigned[i] && distance[a][i] < min) {
114  c = i;
115  min = distance[a][i];
116  }
117  }
118  clusters[currentCluster].push_back(c);
119  assigned[c] = true;
120 
121  currentCluster++;
122  }
123  }
