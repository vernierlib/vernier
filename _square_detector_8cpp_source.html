21  static void _findMarkerContours(const Mat &in, vector<vector<Point2f> > &candidates,
22  vector<vector<Point> > &contoursOut, double minPerimeterRate,
23  double maxPerimeterRate, double accuracyRate,
24  double minCornerDistanceRate, int minDistanceToBorder, int minSize) {
25 
26  CV_Assert(minPerimeterRate > 0 && maxPerimeterRate > 0 && accuracyRate > 0 &&
27  minCornerDistanceRate >= 0 && minDistanceToBorder >= 0);
28 
29  // calculate maximum and minimum sizes in pixels
30  unsigned int minPerimeterPixels =
31  (unsigned int) (minPerimeterRate * max(in.cols, in.rows));
32  unsigned int maxPerimeterPixels =
33  (unsigned int) (maxPerimeterRate * max(in.cols, in.rows));
34 
35  // for aruco3 functionality
36  if (minSize != 0) {
37  minPerimeterPixels = 4 * minSize;
38  }
39 
40  vector<vector<Point> > contours;
41  findContours(in, contours, RETR_LIST, CHAIN_APPROX_NONE);
42  // now filter list of contours
43  for (unsigned int i = 0; i < contours.size(); i++) {
44  // check perimeter
45  if (contours[i].size() < minPerimeterPixels || contours[i].size() > maxPerimeterPixels)
46  continue;
47 
48  // check is square and is convex
49  vector<Point> approxCurve;
50  approxPolyDP(contours[i], approxCurve, double(contours[i].size()) * accuracyRate, true);
51  if (approxCurve.size() != 4 || !isContourConvex(approxCurve)) continue;
52 
53  // check min distance between corners
54  double minDistSq = max(in.cols, in.rows) * max(in.cols, in.rows);
55  for (int j = 0; j < 4; j++) {
56  double d = (double) (approxCurve[j].x - approxCurve[(j + 1) % 4].x) *
57  (double) (approxCurve[j].x - approxCurve[(j + 1) % 4].x) +
58  (double) (approxCurve[j].y - approxCurve[(j + 1) % 4].y) *
59  (double) (approxCurve[j].y - approxCurve[(j + 1) % 4].y);
60  minDistSq = min(minDistSq, d);
61  }
62  double minCornerDistancePixels = double(contours[i].size()) * minCornerDistanceRate;
63  if (minDistSq < minCornerDistancePixels * minCornerDistancePixels) continue;
64 
65  // check if it is too near to the image border
66  bool tooNearBorder = false;
67  for (int j = 0; j < 4; j++) {
68  if (approxCurve[j].x < minDistanceToBorder || approxCurve[j].y < minDistanceToBorder ||
69  approxCurve[j].x > in.cols - 1 - minDistanceToBorder ||
70  approxCurve[j].y > in.rows - 1 - minDistanceToBorder)
71  tooNearBorder = true;
72  }
73  if (tooNearBorder) continue;
74 
75  // if it passes all the test, add to candidates vector
76  vector<Point2f> currentCandidate;
77  currentCandidate.resize(4);
78  for (int j = 0; j < 4; j++) {
79  currentCandidate[j] = Point2f((float) approxCurve[j].x, (float) approxCurve[j].y);
80  }
81  candidates.push_back(currentCandidate);
82  contoursOut.push_back(contours[i]);
83  }
84  }

99  static void _detectInitialCandidates(const Mat &grey, vector<vector<Point2f> > &candidates,
100  vector<vector<Point> > &contours,
101  const SquareDetectorParameters &params) {
102 
103  CV_Assert(params.adaptiveThreshWinSizeMin >= 3 && params.adaptiveThreshWinSizeMax >= 3);
104  CV_Assert(params.adaptiveThreshWinSizeMax >= params.adaptiveThreshWinSizeMin);
105  CV_Assert(params.adaptiveThreshWinSizeStep > 0);
106 
107  // number of window sizes (scales) to apply adaptive thresholding
108  int nScales = (params.adaptiveThreshWinSizeMax - params.adaptiveThreshWinSizeMin) /
109  params.adaptiveThreshWinSizeStep + 1;
110 
111  vector<vector<vector<Point2f> > > candidatesArrays((size_t) nScales);
112  vector<vector<vector<Point> > > contoursArrays((size_t) nScales);
113 
115  parallel_for_(Range(0, nScales), [&](const Range & range) {
116  const int begin = range.start;
117  const int end = range.end;
118 
119  for (int i = begin; i < end; i++) {
120  int currScale = params.adaptiveThreshWinSizeMin + i * params.adaptiveThreshWinSizeStep;
121  // threshold
122  Mat thresh;
123  _threshold(grey, thresh, currScale, params.adaptiveThreshConstant);
124 
125  // detect rectangles
126  _findMarkerContours(thresh, candidatesArrays[i], contoursArrays[i],
127  params.minMarkerPerimeterRate, params.maxMarkerPerimeterRate,
128  params.polygonalApproxAccuracyRate, params.minCornerDistanceRate,
129  params.minDistanceToBorder, params.minSideLengthCanonicalImg);
130  }
131  });
132  // join candidates
133  for (int i = 0; i < nScales; i++) {
134  for (unsigned int j = 0; j < candidatesArrays[i].size(); j++) {
135  candidates.push_back(candidatesArrays[i][j]);
136  contours.push_back(contoursArrays[i][j]);
137  }
138  }

144  static void _reorderCandidatesCorners(vector<vector<Point2f> > &candidates) {
145 
146  for (unsigned int i = 0; i < candidates.size(); i++) {
147  double dx1 = candidates[i][1].x - candidates[i][0].x;
148  double dy1 = candidates[i][1].y - candidates[i][0].y;
149  double dx2 = candidates[i][2].x - candidates[i][0].x;
150  double dy2 = candidates[i][2].y - candidates[i][0].y;
151  double crossProduct = (dx1 * dy2) - (dy1 * dx2);
152 
153  if (crossProduct < 0.0) { // not clockwise direction
154  swap(candidates[i][1], candidates[i][3]);
155  }
156  }
157  }

159  static float getAverageModuleSize(const vector<Point2f>& markerCorners, int markerSize, int markerBorderBits) {
160  float averageArucoModuleSize = 0.f;

205  float static inline getAverageDistance(const std::vector<Point2f>& marker1, const std::vector<Point2f>& marker2) {
206  float minDistSq = std::numeric_limits<float>::max();
207  // fc is the first corner considered on one of the markers, 4 combinations are possible
208  for (int fc = 0; fc < 4; fc++) {
209  float distSq = 0;
210  for (int c = 0; c < 4; c++) {
211  // modC is the corner considering first corner is fc
212  int modC = (c + fc) % 4;
213  distSq += normL2Sqr<float>(marker1[modC] - marker2[c]);
214  }
215  distSq /= 4.f;
216  minDistSq = min(minDistSq, distSq);
217  }
218  return sqrt(minDistSq);
219  }

258  vector<MarkerCandidateTree>
259  filterTooCloseCandidates(vector<vector<Point2f> > &candidates, vector<vector<Point> > &contours) {
260  CV_Assert(detectorParams.minMarkerDistanceRate >= 0.);
261  vector<MarkerCandidateTree> candidateTree(candidates.size());
262  for (size_t i = 0ull; i < candidates.size(); i++) {
263  candidateTree[i] = MarkerCandidateTree(std::move(candidates[i]), std::move(contours[i]));
264  }
265  candidates.clear();
266  contours.clear();
267 
268  // sort candidates from big to small
269  std::stable_sort(candidateTree.begin(), candidateTree.end());
270  // group index for each candidate
271  vector<int> groupId(candidateTree.size(), -1);
272  vector<vector<size_t> > groupedCandidates;
273  vector<bool> isSelectedContours(candidateTree.size(), true);
274 
275  size_t countSelectedContours = 0ull;
276  for (size_t i = 0ull; i < candidateTree.size(); i++) {
277  for (size_t j = i + 1ull; j < candidateTree.size(); j++) {
278  float minDist = getAverageDistance(candidateTree[i].corners, candidateTree[j].corners);
279  // if mean distance is too low, group markers
280  // the distance between the points of two independent markers should be more than half the side of the marker
281  // half the side of the marker = (perimeter / 4) * 0.5 = perimeter * 0.125
282  if (minDist < candidateTree[j].perimeter * (float) detectorParams.minMarkerDistanceRate) {
283  isSelectedContours[i] = false;
284  isSelectedContours[j] = false;
285  // i and j are not related to a group
286  if (groupId[i] < 0 && groupId[j] < 0) {
287  // mark candidates with their corresponding group number
288  groupId[i] = groupId[j] = (int) groupedCandidates.size();
289  // create group
290  groupedCandidates.push_back({i, j});
291  }// i is related to a group
292  else if (groupId[i] > -1 && groupId[j] == -1) {
293  int group = groupId[i];
294  groupId[j] = group;
295  // add to group
296  groupedCandidates[group].push_back(j);
297  }// j is related to a group
298  else if (groupId[j] > -1 && groupId[i] == -1) {
299  int group = groupId[j];
300  groupId[i] = group;
301  // add to group
302  groupedCandidates[group].push_back(i);
303  }
304  }
305  }
306  countSelectedContours += isSelectedContours[i];
307  }
308 
309  for (vector<size_t>& grouped : groupedCandidates) {
310  if (detectorParams.detectInvertedMarker) // if detectInvertedMarker choose smallest contours
311  std::stable_sort(grouped.begin(), grouped.end(), [](const size_t & a, const size_t & b) {
312  return a > b;
313  });
314  else // if detectInvertedMarker==false choose largest contours
315  std::stable_sort(grouped.begin(), grouped.end());
316  size_t currId = grouped[0];
317  isSelectedContours[currId] = true;
318  for (size_t i = 1ull; i < grouped.size(); i++) {
319  size_t id = grouped[i];
320  float dist = getAverageDistance(candidateTree[id].corners, candidateTree[currId].corners);
321  //float moduleSize = getAverageModuleSize(candidateTree[id].corners, dictionary.markerSize, detectorParams.markerBorderBits);
322  float moduleSize = getAverageModuleSize(candidateTree[id].corners, detectorParams.markerSize, detectorParams.markerBorderBits);
323  if (dist > detectorParams.minGroupDistance * moduleSize) {
324  currId = id;
325  candidateTree[grouped[0]].closeContours.push_back(candidateTree[id]);
326  }
327  }
328  }
329 
330  vector<MarkerCandidateTree> selectedCandidates(countSelectedContours + groupedCandidates.size());
331  countSelectedContours = 0ull;
332  for (size_t i = 0ull; i < candidateTree.size(); i++) {
333  if (isSelectedContours[i]) {
334  selectedCandidates[countSelectedContours] = std::move(candidateTree[i]);
335  countSelectedContours++;
336  }
337  }
338 
339  // find hierarchy in the candidate tree
340  for (int i = (int) selectedCandidates.size() - 1; i >= 0; i--) {
341  for (int j = i - 1; j >= 0; j--) {
342  if (checkMarker1InMarker2(selectedCandidates[i].corners, selectedCandidates[j].corners)) {
343  selectedCandidates[i].parent = j;
344  selectedCandidates[j].depth = max(selectedCandidates[j].depth, selectedCandidates[i].depth + 1);
345  break;
346  }
347  }
348  }
349  return selectedCandidates;
350  }
