63  std::vector<Pose> PeriodicPatternDetector::getAll3DPoses(int id) {
64 
65  double u1 = plane1.getA() * 2.0 * PI / (pow(plane1.getA(), 2) + pow(plane1.getB(), 2));
66  double v1 = plane1.getB() * 2.0 * PI / (pow(plane1.getA(), 2) + pow(plane1.getB(), 2));
67  double u2 = plane2.getA() * 2.0 * PI / (pow(plane2.getA(), 2) + pow(plane2.getB(), 2));
68  double v2 = plane2.getB() * 2.0 * PI / (pow(plane2.getA(), 2) + pow(plane2.getB(), 2));
69 
70  double alpha = plane1.getAngle();
71 
72  double b = (pow(u1, 2) + pow(v1, 2)) / pow(physicalPeriod, 2);
73  double g = pow(-u2 * sin(alpha) + v2 * cos(alpha), 2) / pow(physicalPeriod, 2);
74  double d = pow(u2 * cos(alpha) + v2 * sin(alpha), 2) / pow(physicalPeriod, 2);
75 
76  double s_2 = (b + g + d + sqrt(pow(b + g + d, 2) - 4.0 * b * g)) / 2.0;
77 
78  double gamma11 = acos(sqrt(g / s_2));
79  double gamma12 = -acos(sqrt(g / s_2));
80  double gamma21 = acos(sqrt(g / s_2));
81  double gamma22 = -acos(sqrt(g / s_2));
82 
83  double beta11 = acos(u1 / sqrt(s_2) / physicalPeriod / cos(alpha));
84  double beta12 = acos(u1 / sqrt(s_2) / physicalPeriod / cos(alpha));
85  double beta21 = -acos(u1 / sqrt(s_2) / physicalPeriod / cos(alpha));
86  double beta22 = -acos(u1 / sqrt(s_2) / physicalPeriod / cos(alpha));
87 
88  double x = -plane1.getPosition(physicalPeriod, 0.0, 0.0, periodShift1);
89  double y = -plane2.getPosition(physicalPeriod, 0.0, 0.0, periodShift2);
90 
91  alpha = angleInPiPi(plane2.getAngle() - PI / 2);
92 
93  std::vector<Pose> poseVec = {Pose(x, y, 0, alpha, beta11, gamma11, 1.0 / sqrt(s_2)),
94  Pose(x, y, 0, alpha, beta11, gamma12, 1.0 / sqrt(s_2)),
95  Pose(x, y, 0, alpha, beta21, gamma21, 1.0 / sqrt(s_2)),
96  Pose(x, y, 0, alpha, beta22, gamma22, 1.0 / sqrt(s_2))};
97 
98  return poseVec;
99  }
