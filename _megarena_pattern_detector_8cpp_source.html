37  void MegarenaPatternDetector::readJSON(const rapidjson::Value& document) {

41  if (document.HasMember("period") && document["period"].IsDouble()) {
42  physicalPeriod = document["period"].GetDouble();
43  PeriodicPatternDetector::setPhysicalPeriod(physicalPeriod);
44  } else {

52  const rapidjson::Value& value = document["bitSequence"][row];

57  }
58  }
59  } else if (document.HasMember("codeSize") && document["codeSize"].IsInt()) {
60  int codeSize = document["codeSize"].GetInt();
61  if (codeSize >= 4 && codeSize <= 12) {
62  MegarenaBitSequence::generate(codeSize, bitSequence);
63  } else {
64  throw Exception("The file is not a valid megarena pattern file, the code size must between 4 and 12.");
65  }
66  } else {
67  throw Exception("The file is not a valid bitmap pattern file, the bitmap is missing or has a wrong format.");
68  }
69 
70  decoding.resize(bitSequence);
71  }

78  void MegarenaPatternDetector::computeAbsolutePose() {
79  double approxPixelPeriod = (plane1.getPixelicPeriod() + plane2.getPixelicPeriod()) / 2.0;
80 
81  int length1 = (array.rows() / (approxPixelPeriod));
82  int length2 = (array.cols() / (approxPixelPeriod));
83 
84  length1++;
85  length2++;
86 
87  if (length1 % 2 == 0) {
88  length1++;
89  }
90  if (length2 % 2 == 0) {
91  length2++;
92  }
93 
94  thumbnail.resize(length1, length2);
95  thumbnail.compute(plane1, plane2, array);
96 
97  Eigen::ArrayXXd sequence1 = thumbnail.getSequence1();
98  Eigen::ArrayXXd sequence2 = thumbnail.getSequence2();
99 
100  int MSB1 = thumbnail.getMSB1();
101  int MSB2 = thumbnail.getMSB2();
102 
103  periodShift1 = decoding.findCodePosition(sequence1, MSB1);
104  periodShift2 = decoding.findCodePosition(sequence2, MSB2);
105 
106  // plane1Save = plane1;
107  // plane2Save = plane2;
108 
109  if (periodShift1 >= 0 && periodShift2 >= 0) {
110  // no change
111  //std::cout<<"code1>0 && code2>0"<<std::endl;
112  } else if (periodShift1 < 0 && periodShift2 >= 0) {
113  std::swap(periodShift1, periodShift2);
114  std::swap(plane1, plane2);
115  plane2.flip();
116  //std::cout<<"code1<0 && code2>0"<<std::endl;
117  thumbnail.rotate270();
118  } else if (periodShift1 >= 0 && periodShift2 < 0) {
119  std::swap(periodShift1, periodShift2);
120  std::swap(plane1, plane2);
121  plane1.flip();
122  //std::cout<<"code1>0 && code2<0"<<std::endl;
123  thumbnail.rotate90();
124  } else {
125  plane1.flip();
126  plane2.flip();
127  //std::cout<<"code1<0 && code2<0"<<std::endl;
128  thumbnail.rotate180();
129  }
130 
131  periodShift1 = std::abs(periodShift1);
132  periodShift2 = std::abs(periodShift2);
133 
134  }
