155  int MegarenaAbsoluteDecoding::findCodePosition(Eigen::ArrayXXd& codeSample, int MSB) {
156  double maximum;
157  int offset = floor(codeSample.rows() / 2);
158  int direction = 1;
159 
160  if (MSB == 0) {
161  codeSample.colwise().reverseInPlace();
162  direction = -1;
163  }
164 
165  //bitSequence = bitSequence.block(0, 0, 1, 100);
166 
167  // convolution with 'same' mode
168  Eigen::ArrayXXd testPeak(1, bitSequence.cols());
169  for (int i = 0; i < bitSequence.cols(); i++) {
170  int jMin;
171  int jMax;
172  if (i >= offset) {
173  jMin = 0;
174  } else {
175  jMin = offset - i;
176  }
177 
178  if (i + offset < bitSequence.cols()) {
179  jMax = codeSample.rows() - 1;
180  } else {
181  jMax = bitSequence.cols() - 1 - i + offset;
182  }
183 
184  testPeak(0, i) = 0;
185  for (int j = jMin; j <= jMax; j++) {
186  testPeak(0, i) += (double) bitSequence(0, i - offset + j) * codeSample(j, 0);
187  }
188  }
189 
190  Eigen::MatrixXd::Index maxRow, maxCol;
191 
192  maximum = testPeak.maxCoeff(&maxRow, &maxCol);
193 
194  return direction * maxCol;
195  }
