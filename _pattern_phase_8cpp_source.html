53  void PatternPhase::compute() {
54  fft.compute(spatial, spectrum);
55 
56  shift(spectrum, spectrumShifted);
57  spectrumFiltered1 = spectrumShifted;
58  spectrumFiltered2 = spectrumShifted;
59 
60  Eigen::ArrayXXd magnitude = spectrumShifted.abs();
61  peaksSearch(magnitude, mainPeak1, mainPeak2);
62 
63  // Compute first plane phase from peak 1
64  applyGaussianFilter(spectrumFiltered1, mainPeak1(1), mainPeak1(0), sigma);
65 
66  ifft.compute(spectrumFiltered1, phase1);

75  // Compute second plase from peak 2

87  void PatternPhase::peaksSearch(Eigen::ArrayXXd& source, Eigen::Vector3d& mainPeak1, Eigen::Vector3d& mainPeak2) {
88  mainPeak1.setConstant(-1);
89  mainPeak2.setConstant(-1);
90 
91  applyBandPassCut(source, minFrequency, maxFrequency);
92 
93  // Filtre moyenneur 3x3 -> Ã  faire avant dans le domaine spatial
94  // puis utiliser source.maxCoeff(&row, &col);
95 
96  double maxValue = -1.0;
97  for (int col = 1; col < source.cols() - 1; col++) {
98  for (int row = source.rows() / 2; row < source.rows() - 1; row++) {
99  double mean = (source(row, col) + source(row - 1, col) + source(row, col - 1) + source(row + 1, col) + source(row, col + 1)) / 5.0;
100  if (mean > maxValue) {
101  maxValue = mean;
102  mainPeak1.x() = col;
103  mainPeak1.y() = row;
104  mainPeak1.z() = mean / source.cols() / source.rows();
105  }
106  }
107  }
108 
109  if (mainPeak1.z() > minPeakPower) {
110  double vx = mainPeak1.x() - source.cols() / 2;
111  double vy = mainPeak1.y() - source.rows() / 2;
112  double distance = std::hypot(vx, vy);
113 
114  double centerAngle = std::atan2(vy, vx);
115  double widthAngle = 2.0 * std::atan2(3.0 * sigma, distance);
116  applyAngularCut(source, centerAngle, widthAngle);
117 
118  maxValue = -1;
119  for (int col = 1; col < source.cols() - 1; col++) {
120  for (int row = source.rows() / 2; row < source.rows() - 1; row++) {
121  double mean = (source(row, col) + source(row - 1, col) + source(row, col - 1) + source(row + 1, col) + source(row, col + 1)) / 5.0;
122  if (mean > maxValue) {
123  maxValue = mean;
124  mainPeak2.x() = col;
125  mainPeak2.y() = row;
126  mainPeak2.z() = mean / source.cols() / source.rows();
127  }
128  }
129  }
130 
131  if (mainPeak1.x() < mainPeak2.x()) {
132  std::swap(mainPeak1, mainPeak2);
133  }
134 
135  }
136  }
