29  void BitmapThumbnail::compute(const Eigen::ArrayXXd & array, PhasePlane plane1, PhasePlane plane2) {
30  numberWhiteDots.fill(0);
31  cumulWhiteDots.fill(0);
32 
33 #pragma omp parallel for
34  for (int col = 0; col < array.cols(); col++) {
35  for (int row = 0; row < array.rows(); row++) {
36 
37  double phaseCol = plane1.getPhase(row - array.rows() / 2, col - array.cols() / 2);
38  double phaseRow = plane2.getPhase(row - array.rows() / 2, col - array.cols() / 2);
39 
40  int phaseIteration1 = round(phaseCol / PI) + thumbnail.cols / 2;
41  int phaseIteration2 = round(phaseRow / PI) + thumbnail.rows / 2;
42 
43  if (phaseIteration1 < numberWhiteDots.rows() && phaseIteration2 < numberWhiteDots.cols() && phaseIteration1 >= 0 && phaseIteration2 >= 0) {
44  if ((abs(std::fmod(phaseCol, PI)) <= DELTA_PHASE || abs(std::fmod(phaseCol, PI)) >= PI - DELTA_PHASE) && (abs(std::fmod(phaseRow, PI)) <= DELTA_PHASE || abs(std::fmod(phaseRow, PI)) >= PI - DELTA_PHASE)) {
45  numberWhiteDots(phaseIteration1, phaseIteration2) += 1;
46  cumulWhiteDots(phaseIteration1, phaseIteration2) += array(row, col);
47  }
48  }
49  }
50  }

52  double meanBackground = 0.0;
53  int countBackground = 0;
54  double meanForeground = 0.0;
55  int countForeground = 0;
56 #pragma omp parallel for
57  for (int row = 0; row < thumbnail.rows; row++) {
58  for (int col = 0; col < thumbnail.cols; col++) {
59  thumbnail.at<unsigned char>(row, col) = (unsigned char) (255 * cumulWhiteDots(col, row) / numberWhiteDots(col, row));
60  if (row % 2 == 0 && col % 2 == 0) {
61  meanForeground += thumbnail.at<unsigned char>(row, col);
62  countForeground++;
63  } else {
64  meanBackground += thumbnail.at<unsigned char>(row, col);
65  countBackground++;
66  }
67  }
68  }
69  meanBackground /= countBackground;
70  meanForeground /= countForeground;
71 
72  unsigned char threshold = (unsigned char) ((meanBackground + meanForeground) / 2);
73  cv::threshold(thumbnail, binaryThumbnail, threshold, 255, cv::THRESH_BINARY);
