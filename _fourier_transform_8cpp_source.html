38  if (nRows <= 0 || nCols <= 0) {
39  throw Exception("Can't resize a FourierTransform with rows<=0 or cols<=0");
40  } else if (nRows != this->nRows || nCols != this->nCols || sign != this->sign) {
41  if (plan != NULL) {
42  fftw_destroy_plan(plan);
43  }
44 
45  this->nRows = nRows;
46  this->nCols = nCols;
47  this->sign = sign;
48 
49  fftw_complex* in = (fftw_complex*) fftw_malloc(sizeof (fftw_complex) * nRows * nCols);
50  fftw_complex* out = (fftw_complex*) fftw_malloc(sizeof (fftw_complex) * nRows * nCols);
51 
52  if (nRows == 1 || nCols == 1) {
53  plan = fftw_plan_dft_1d(nRows * nCols, in, out, sign, FFTW_MEASURE);

121  void FourierTransform::resize(int nRows, int nCols, int sign) {
122  if (nRows <= 0 || nCols <= 0) {
123  throw Exception("Can't resize a FourierTransform with rows<=0 or cols<=0");
124  } else if (((nRows & (nRows - 1)) != 0) || ((nCols & (nCols - 1)) != 0)) {
125  throw Exception("The dimensions of a FourierTransform must be a power of two");
126  } else if (nRows != this->nRows || nCols != this->nCols || sign != this->sign) {
127  if (data != NULL) {
128  free(data);
129  }
130  if (workArea != NULL) {
131  free(workArea);
132  }
133  if (bitReversal != NULL) {
134  free(bitReversal);
135  }
136  if (cosSinTable != NULL) {
137  free(cosSinTable);
138  }
139 
140  this->nRows = nRows;
141  this->nCols = nCols;
142  this->sign = sign;
143 
144  data = (double**) malloc(sizeof (double*) * nCols);
145  workArea = NULL;
146  int n = 2 + (int) sqrt(std::max(nRows, nCols));
147  bitReversal = (int*) malloc(sizeof (int) * n);
148  for (int i = 0; i < n; i++) {
149  bitReversal[i] = 0;
150  }
151  n = std::max(nRows, nCols) / 2;
152  cosSinTable = (double*) malloc(sizeof (double) * n);
153  }
154  }

156  void FourierTransform::compute(const Eigen::ArrayXXcd& in, Eigen::ArrayXXcd& out) {
157  resize(in.rows(), in.cols(), sign);
158  out.resize(nRows, nCols);
159  out = in;
160  // Eigen is column major but Ooura's fft is row major
161  for (int j = 0; j < nCols; j++) {
162  data[j] = (double*) (&out(0, j));
163  }
164  cdft2d(nCols, 2*nRows, sign, data, workArea, bitReversal, cosSinTable);
165  out = out.conjugate();
166  }
