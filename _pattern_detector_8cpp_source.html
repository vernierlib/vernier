56  void PatternDetector::compute(const cv::Mat & image) {
57  ASSERT_MSG(image.cols > 0 && image.rows > 0, "The image is empty.")
58  if (image.depth() == CV_8U) {
59  if (image.channels() > 1) {
60  cv::cvtColor(image, image8U, cv::COLOR_BGR2GRAY);
61  } else {
62  image8U = image;
63  }
64  image8U.convertTo(image32F, CV_32F, 1.0 / 255);
65  image8U.convertTo(image64F, CV_64F, 1.0 / 255);
66  cv2eigen(image64F, array);
67  } else if (image.depth() == CV_16U) {
68  if (image.channels() > 1) {
69  cv::cvtColor(image, image64F, cv::COLOR_BGR2GRAY);
70  } else {
71  image64F = image;
72  }
73  image64F.convertTo(image64F, CV_64F, 1.0 / 65535);
74  image64F.convertTo(image32F, CV_32F);
75  image64F.convertTo(image8U, CV_8U, 255);
76  cv2eigen(image64F, array);
77  } else if (image.depth() == CV_32F) {
78  if (image.channels() > 1) {
79  cv::cvtColor(image, image32F, cv::COLOR_BGR2GRAY);
80  } else {
81  image32F = image;
82  }
83  image32F.convertTo(image8U, CV_8U, 255);
84  image32F.convertTo(image64F, CV_64F);
85  cv2eigen(image64F, array);
86  } else {
87  throw Exception("The image must be 8-bit, 16-bit or 32-bit floating-point.");
88  }
89  computeImage();
90  }
